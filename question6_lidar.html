import numpy as np 
import csv

import pylab as pl


data_lid1, data_lid2 = [[],[],[],[],[],[],[],[]],[[],[],[],[],[],[],[],[]] 

def read_data():
	prefix = ['Lidar_1_', 'Lidar_2_']
	suffix = '.csv'
	for i in range(0,8):
		if i==0:
			prefix[0] = prefix[0] + '00'
			prefix[1] = prefix[1] + '00'
		elif i<3:
			prefix[0] = prefix[0] + '0'
			prefix[1] = prefix[1] + '0'
		filename_1 = prefix[0] + str(i*45) + suffix
		filename_2 = prefix[1] + str(i*45) + suffix		
		prefix[0], prefix[1] = 'Lidar_1_', 'Lidar_2_'
		# print (filename_1, " \t ", filename_2)
		with open(filename_1, 'r') as datafile1:
			data1 = csv.reader(datafile1)
			for d in data1:
				dist= int(d[0])
				data_lid1[i].append(dist)
		with open(filename_2, 'r') as datafile1:
			data1 = csv.reader(datafile1)
			for d in data1:
				dist= int(d[0])
				data_lid2[i].append(dist)


def create_map_lidar1():

	# xx1 = [np.mean(np.asarray(data_lid1[i]))*cos(radians(i*45+90)) for i in range(0,8)]
	# yy1 = [np.mean(np.asarray(data_lid1[i]))*sin(radians(i*45+90)) for i in range(0,8)]
	xx1, yy1 = [], []

	xx1.insert(0,0) 
	yy1.insert(0,0)

	for i in range(0,8):
		xx1.append(np.mean(np.asarray(data_lid1[i]))*cos(radians(i*45+90)))
		xx1.append(0)

	for i in range(0,8):
		yy1.append(np.mean(np.asarray(data_lid1[i]))*sin(radians(i*45+90)))
		yy1.append(0)

	# print (xx1)
	# print (yy1)

	number = [0,1,2,3,4,5,6,7,8]
	c=0
	# img = pl.imread('Selected P85 Map.png')
	# pl.imshow(img, extent=[-1500, 1500, -1000, 400])
	pl.plot(xx1, yy1, "-o", linewidth=3)
	for x, y in zip(xx1, yy1):
	    pl.text(x, y, 'x:' + str(float("{0:.2f}".format(x))) + '\n' + 'y:' + str(float("{0:.2f}".format(y))) , color="red", fontsize=8)  # 'Point:' + str(number[c]) + '\n' + 
	    c+=1
	pl.grid(True)
	pl.xlabel("x")
	pl.title("Robot 1 Mapping")
	pl.show()
