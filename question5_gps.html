import numpy as np 
import csv
import math

from math import sin, cos, atan2, sqrt, degrees, radians

#Declaring global variables

data_rover1, data_rover2, data_master = [], [], []
totalDeg_lat_r1, totalDeg_lat_r2, totalDeg_lat_mas, totalDeg_lon_r1, totalDeg_lon_r2, totalDeg_lon_mas = [], [], [], [], [], []

avg_lat_rover1, avg_lon_rover1, avg_lat_rover2, avg_lon_rover2, avg_lat_master, avg_lon_master = 0,0,0,0,0,0

e_mtr_lat_r1, e_mtr_lon_r1, e_mtr_lat_r2, e_mtr_lon_r2, e_mtr_lat_mas, e_mtr_lon_mas = [], [], [], [], [], []
e_deg_lat_r1, e_deg_lon_r1, e_deg_lat_r2, e_deg_lon_r2, e_deg_lat_mas, e_deg_lon_mas = [], [], [], [], [], []


# Reading Data from the files

def read_data():
	with open('Rover_1.csv', 'r') as datafile1:
		data1 = csv.reader(datafile1)
		for d in data1:
			time, lat, lon = d[0], d[1], d[3]
			data_rover1.append(tuple((time, lat, lon)))
		del data_rover1[0]	
	with open('Rover_2.csv', 'r') as datafile1:
		data1 = csv.reader(datafile1)
		for d in data1:
			time, lat, lon = d[0], d[1], d[3]			
			data_rover2.append(tuple((time, lat, lon)))
		del data_rover2[0]	
	with open('Master.csv', 'r') as datafile1:
		data1 = csv.reader(datafile1)
		for d in data1:
			time, lat, lon = d[0], d[1], d[3]
			data_master.append(tuple((time, lat, lon)))
		del data_master[0]	
			
# Converting GPS readings to degrees

def gps_to_degrees():

	for tt, la, lo in data_rover1:
		totalDeg_lat_r1.append( int(float(la)/100) + (float(la)- 100*int(float(la)/100))/60) 
		totalDeg_lon_r1.append( int(float(lo)/100) + (float(lo)- 100*int(float(lo)/100))/60) 
	for tt, la, lo in data_rover2:
		totalDeg_lat_r2.append( int(float(la)/100) + (float(la)- 100*int(float(la)/100))/60) 
		totalDeg_lon_r2.append( int(float(lo)/100) + (float(lo)- 100*int(float(lo)/100))/60) 
	for tt, la, lo in data_master:
		totalDeg_lat_mas.append( int(float(la)/100) + (float(la)- 100*int(float(la)/100))/60)
		totalDeg_lon_mas.append( int(float(lo)/100) + (float(lo)- 100*int(float(lo)/100))/60) 

	# print (totalDeg_lat_r1[0])
	# print (totalDeg_lat_r2[0])
	# print (totalDeg_lat_mas[0])	

# Calculating average locations - Part A

def avg_loc_part_A():
	avg_lat_rover1, avg_lon_rover1 = np.mean(np.asarray(totalDeg_lat_r1)), np.mean(np.asarray(totalDeg_lon_r1))
	avg_lat_rover2, avg_lon_rover2 = np.mean(np.asarray(totalDeg_lat_r2)), np.mean(np.asarray(totalDeg_lon_r2))
	avg_lat_master, avg_lon_master = np.mean(np.asarray(totalDeg_lat_mas)), np.mean(np.asarray(totalDeg_lon_mas))

	print ("Question 5: Part A \n")
	print ("Average Location in degrees for Rover1: \t Latitiude = %f \t Longitude = %f" %(avg_lat_rover1, avg_lon_rover1))
	print ("Average Location in degrees for Rover2: \t Latitiude = %f \t Longitude = %f" %(avg_lat_rover2, avg_lon_rover2))
	print ("Average Location in degrees for Master: \t Latitiude = %f \t Longitude = %f \n" %(avg_lat_master, avg_lon_master))	

	return avg_lat_rover1, avg_lon_rover1, avg_lat_rover2, avg_lon_rover2, avg_lat_master, avg_lon_master	

def error_degrees_part_B(avg_lat_rover1, avg_lon_rover1, avg_lat_rover2, avg_lon_rover2, avg_lat_master, avg_lon_master):

	for i in range(0, len(totalDeg_lat_r1)):
		e_deg_lat_r1.append((totalDeg_lat_r1[i]-avg_lat_rover1))
		e_deg_lon_r1.append(totalDeg_lon_r1[i]-avg_lon_rover1)
	
	for i in range(0, len(totalDeg_lat_r2)):
		e_deg_lat_r2.append(totalDeg_lat_r2[i]-avg_lat_rover2)
		e_deg_lon_r2.append(totalDeg_lon_r2[i]-avg_lon_rover2)

	for i in range(0, len(totalDeg_lat_mas)):
		e_deg_lat_mas.append(totalDeg_lat_mas[i]-avg_lat_master)
		e_deg_lon_mas.append(totalDeg_lon_mas[i]-avg_lon_master)

# Calculating standard deviations of errors - Part B 

def error_std_metres_part_B():
	''' For average location of Rover 1 at Latitiude = 42.035322 - 1 degree = 111073.94016055862 | Longitude = 71.696464 - 1 degree = 82804.89671211915  
		Using this value of 1 degree, we convert the various errors in degrees to meters'''
	e_mtr_lat_r1 = [i*111073.94016055862 for i in e_deg_lat_r1]
	e_mtr_lon_r1 = [i*82804.89671211915 for i in e_deg_lon_r1]
	
	''' For average location of Rover 2 at Latitiude = 42.035235 - 1 degree = 111073.93847103874 | Longitude = 71.696425 - 1 degree = 82805.00964398625  
		Using this value of 1 degree, we convert the various errors in degrees to meters'''
	e_mtr_lat_r2 = [i*111073.93847103874 for i in e_deg_lat_r2]
	e_mtr_lon_r2 = [i*82805.00964398625 for i in e_deg_lon_r2]

	''' For average location of Master at Latitiude = 42.035395 - 1 degree = 111073.94157820214 | Longitude = 71.696533 - 1 degree = 82804.80195304807 
		Using this value of 1 degree, we convert the various errors in degrees to meters'''
	e_mtr_lat_mas = [i*111073.94157820214  for i in e_deg_lat_mas]
	e_mtr_lon_mas = [i*82804.80195304807 for i in e_deg_lat_mas]


	std_err_mtr_lat_r1 = np.std(np.asarray(e_deg_lat_r1))*111073.94016055862
	std_err_mtr_lon_r1 = np.std(np.asarray(e_deg_lon_r1))*82804.89671211915

	std_err_mtr_lat_r2 = np.std(np.asarray(e_deg_lat_r2))*111073.93847103874 
	std_err_mtr_lon_r2 = np.std(np.asarray(e_deg_lon_r2))*82805.00964398625

	std_err_mtr_lat_mas = np.std(np.asarray(e_deg_lat_mas))*111073.94157820214
	std_err_mtr_lon_mas = np.std(np.asarray(e_deg_lon_mas))*82804.80195304807

	print ("Question 5: Part B \n")
	print ("Standard deviation of error in meters for Rover1: Latitiude: %f meters | Longitude: %f meters" %(std_err_mtr_lat_r1, std_err_mtr_lon_r1))
	print ("Standard deviation of error in meters for Rover2: Latitiude: %f meters | Longitude: %f meters" %(std_err_mtr_lat_r2, std_err_mtr_lon_r2))
	print ("Standard deviation of error in meters for Master: Latitiude: %f meters | Longitude: %f meters \n" %(std_err_mtr_lat_mas, std_err_mtr_lon_mas))

	return e_mtr_lat_r1, e_mtr_lon_r1, e_mtr_lat_r2, e_mtr_lon_r2, e_mtr_lat_mas, e_mtr_lon_mas 

# Calculating distances between the rovers and the master station

def distance_rover_master_partC(avg_lat_rover1, avg_lon_rover1, avg_lat_rover2, avg_lon_rover2, avg_lat_master, avg_lon_master):

		'''	We consider the master position latitude and longitude position to be standard for degree to meter conversion
			Latitiude = 42.035235 - 1 degree = 111073.94157820214 | Longitude = 71.696425 - 1 degree = 82804.80195304807'''


		# This is using the cartesian coordinate method of finding distance between two points -  NEGLECTED	
		# dist_lat_mtr_r1_mas = (avg_lat_master*111073.94157820214 - avg_lat_rover1*111073.94016055862) 
		# dist_lon_mtr_r1_mas = (avg_lon_master*82804.80195304807 - avg_lon_rover1*82804.89671211915)
		
		# dist_mas_r1 = math.sqrt(dist_lat_mtr_r1_mas**2 + dist_lon_mtr_r1_mas**2)


		# Using the technique suggested by the website - https://andrew.hedges.name/experiments/haversine/
		dlon = avg_lon_master - avg_lon_rover1 
		dlat = avg_lat_master - avg_lat_rover1

		a = (sin(radians(dlat/2)))**2 + cos(radians(avg_lat_rover1)) * cos(radians(avg_lat_master)) * (sin(radians(dlon/2)))**2 
		c = 2 * atan2( sqrt(a), sqrt(1-a) ) 
		d = 6373 * c *1000														# (where R is the radius of the Earth)		

		# This is using the cartesian coordinate method of finding distance between two points -  NEGLECTED
		# dist_lat_mtr_r2_mas = (avg_lat_master*111073.94157820214  - avg_lat_rover2*111073.93847103874)
		# dist_lon_mtr_r2_mas = (avg_lon_master*82804.80195304807 - avg_lon_rover2*82805.00964398625)

		# dist_mas_r2 = math.sqrt(dist_lat_mtr_r2_mas**2 + dist_lon_mtr_r2_mas**2)

		# Using the technique suggested by the website - https://andrew.hedges.name/experiments/haversine/
		ddlon = avg_lon_master - avg_lon_rover2 
		ddlat = avg_lat_master - avg_lat_rover2

		aa = (sin(radians(ddlat/2)))**2 + cos(radians(avg_lat_rover2)) * cos(radians(avg_lat_master)) * (sin(radians(ddlon/2)))**2 
		cc = 2 * atan2( sqrt(aa), sqrt(1-aa) ) 
		dd = 6373 * cc *1000														# (where R is the radius of the Earth)		

		print ("Question 5: Part C \n")
		print ("Distance between Rover1 and Master: %f meters" %(d))
		print ("Distance between Rover2 and Master: %f meters\n" %(dd))

# Replacing same time GPS errors

def data_aligned_rover1_master_partD(e_mtr_lat_r1, e_mtr_lon_r1, e_mtr_lat_r2, e_mtr_lon_r2, e_mtr_lat_mas, e_mtr_lon_mas):

	corrected_error_lat_mtr_r1, corrected_error_lon_mtr_r1 = e_mtr_lat_r1, e_mtr_lon_r1	
	corrected_error_lat_mtr_r2, corrected_error_lon_mtr_r2 = e_mtr_lat_r2, e_mtr_lon_r2	

	for i in range(0, len(list(zip(*data_rover1)[0]))-1):
		if data_rover1[i][0] in list(zip(*data_master))[0]:
			pty = zip(*data_master)[0].index(data_rover1[i][0])
			# print (pty, " ----", i)								# Matching time indices

			corrected_error_lat_mtr_r1[i] = e_mtr_lat_r1[i] - e_mtr_lat_mas[pty]
			corrected_error_lon_mtr_r1[i] = e_mtr_lon_r1[i] - e_mtr_lon_mas[pty]

	new_std_err_lat_mtr_r1 = np.std(np.asarray(corrected_error_lat_mtr_r1))
	new_std_err_lon_mtr_r1 = np.std(np.asarray(corrected_error_lon_mtr_r1))

	for j in range(0, len(list(zip(*data_rover2)[0]))-1):
		if data_rover2[j][0] in list(zip(*data_master))[0]:
			ptya = zip(*data_master)[0].index(data_rover2[j][0])
			# print (ptya, " ----", j)								# Matching time indices

			corrected_error_lat_mtr_r2[j] = e_mtr_lat_r2[j] - e_mtr_lat_mas[ptya]
			corrected_error_lon_mtr_r2[j] = e_mtr_lon_r2[j] - e_mtr_lon_mas[ptya]

	new_std_err_lat_mtr_r2 = np.std(np.asarray(corrected_error_lat_mtr_r2))
	new_std_err_lon_mtr_r2 = np.std(np.asarray(corrected_error_lon_mtr_r2))

	print ("Question 5: Part D \n")
	print("Corrected Error Standard Deviation for Rover1:	Latitiude: %f | Longitude: %f " %(new_std_err_lat_mtr_r1, new_std_err_lon_mtr_r1))
	print("Corrected Error Standard Deviation for Rover2:	Latitiude: %f | Longitude: %f \n" %(new_std_err_lat_mtr_r2, new_std_err_lon_mtr_r2))
