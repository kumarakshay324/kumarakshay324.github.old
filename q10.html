def lidar_8a_bias_std_var():

	#Mean
	lidar_2m_HARD_mean = np.mean(data_lidar_2m_HARD)
	lidar_2m_SOFT_mean = np.mean(data_lidar_2m_SOFT)
	print ("Lidar Mean for 2m_HARD range is %f and for 2m_SOFT range is %f" %(lidar_2m_HARD_mean,lidar_2m_SOFT_mean))\

	#Bias 
	lidar_2m_HARD_bias = lidar_2m_HARD_mean - 200.0;
	lidar_2m_SOFT_bias = lidar_2m_SOFT_mean - 200.0;
	print ("Lidar Bias for 2m_HARD range is %f and for 2m_SOFT range is %f" %(lidar_2m_HARD_bias,lidar_2m_SOFT_bias))

	#Variance
	lidar_2m_HARD_var = np.var(data_lidar_2m_HARD);
	lidar_2m_SOFT_var = np.var(data_lidar_2m_SOFT);
	print ("Lidar Variance for 2m_HARD range is %f and for 2m_SOFT range is %f" %(lidar_2m_HARD_var,lidar_2m_SOFT_var))

	lidar_2m_HARD_std = np.std(data_lidar_2m_HARD)
	lidar_2m_SOFT_std = np.std(data_lidar_2m_SOFT)
	print ("Lidar Standard Deviation for 2m_HARD range is %f and for 2m_SOFT range is %f\n" %(lidar_2m_HARD_std,lidar_2m_SOFT_std))

	return lidar_2m_HARD_var, lidar_2m_SOFT_var, lidar_2m_HARD_bias, lidar_2m_SOFT_bias

def ultrasound_8b_bias_std_var():

	#Mean
	ultrasound_2m_HARD_mean = np.mean(data_ultrasound_2m_HARD)
	ultrasound_2m_SOFT_mean = np.mean(data_ultrasound_2m_SOFT)
	print ("Ultrasound Mean for 2m_HARD range is %f and for 2m_SOFT range is %f" %(ultrasound_2m_HARD_mean,ultrasound_2m_SOFT_mean))

	#Bias
	ultrasound_2m_HARD_bias = ultrasound_2m_HARD_mean - 200.0;
	ultrasound_2m_SOFT_bias = ultrasound_2m_SOFT_mean - 200.0;
	print ("Ultrasound Bias for 2m_HARD range is %f and for 2m_SOFT range is %f" %(ultrasound_2m_HARD_bias,ultrasound_2m_SOFT_bias))

	#Variance
	ultrasound_2m_HARD_var = np.var(data_ultrasound_2m_HARD);
	ultrasound_2m_SOFT_var = np.var(data_ultrasound_2m_SOFT);
	print ("Ultrasound Variance for 2m_HARD range is %f and for 2m_SOFT range is %f" %(ultrasound_2m_HARD_var,ultrasound_2m_SOFT_var))

	#Standard Deviation
	ultrasound_2m_HARD_std = np.std(data_ultrasound_2m_HARD)
	ultrasound_2m_SOFT_std = np.std(data_ultrasound_2m_SOFT)

	return ultrasound_2m_HARD_var, ultrasound_2m_SOFT_var, ultrasound_2m_HARD_bias, ultrasound_2m_SOFT_bias
	

def sensor_fusion_8c_2m_HARD_2m_SOFT(lidar_2m_HARD_var, lidar_2m_SOFT_var, lidar_2m_HARD_bias, lidar_2m_SOFT_bias, ultrasound_2m_HARD_var, ultrasound_2m_SOFT_var, ultrasound_2m_HARD_bias, ultrasound_2m_SOFT_bias):

	#Classical Central Limit Theorem
	var_combined_2m_HARD = (lidar_2m_HARD_var * ultrasound_2m_HARD_var)/(lidar_2m_HARD_var + ultrasound_2m_HARD_var)
	var_combined_2m_SOFT = (lidar_2m_SOFT_var * ultrasound_2m_SOFT_var)/(lidar_2m_SOFT_var + ultrasound_2m_SOFT_var)

	# Remember to subtract any bias from each sensor - This HINT IS TO BE USED - DISCOVER ABOUT THIS

	for d1 in range(len(data_lidar_2m_HARD)):

		d1_lidar_bias =  lidar_2m_HARD_bias
		d1_ultrasound_bias = ultrasound_2m_HARD_bias
	
				
		temp1 = var_combined_2m_HARD*(((data_lidar_2m_HARD[d1-1]- d1_lidar_bias) /lidar_2m_HARD_var) + ((data_ultrasound_2m_HARD[d1-1] - d1_ultrasound_bias)/ultrasound_2m_HARD_var))
		data_2m_HARD_fused.append(temp1)

	for d2 in range(len(data_lidar_2m_SOFT)):

		d2_lidar_bias = lidar_2m_SOFT_bias
		d2_ultrasound_bias = ultrasound_2m_SOFT_bias
		
		temp2 = var_combined_2m_SOFT*((data_lidar_2m_SOFT[d2-1]-d2_lidar_bias) /lidar_2m_SOFT_var + (data_ultrasound_2m_SOFT[d2-1]-d2_ultrasound_bias)/ultrasound_2m_SOFT_var)
		data_2m_SOFT_fused.append(temp2)

	optimal_estimate_2m_HARD = np.mean(data_2m_HARD_fused)
	optimal_estimate_2m_SOFT = np.mean(data_2m_SOFT_fused)

	std_2m_HARD = np.std(data_2m_HARD_fused)
	std_2m_SOFT = np.std(data_2m_SOFT_fused)

	print ("Optimal Estimate and Standard Deviation for the 2m_HARD range combined data from the lidar and ultrasound sensor is %f and %f" %(optimal_estimate_2m_HARD, std_2m_HARD))
	print ("Optimal Estimate and Standard Deviation for the 2m_SOFT range combined data from the lidar and ultrasound sensor is %f and %f" %(optimal_estimate_2m_SOFT, std_2m_SOFT))

	print ("Weights for the lidar and ultrasonic sensors for 2m_HARD respectively", var_combined_2m_HARD/lidar_2m_HARD_var, var_combined_2m_HARD/ultrasound_2m_HARD_var)
	print ("Weights for the lidar and ultrasonic sensors for 2m_SOFT respectively", var_combined_2m_SOFT/lidar_2m_SOFT_var, var_combined_2m_SOFT/ultrasound_2m_SOFT_var)
